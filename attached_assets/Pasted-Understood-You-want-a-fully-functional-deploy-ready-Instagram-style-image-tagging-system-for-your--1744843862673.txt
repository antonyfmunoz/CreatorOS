3. JavaScript (script.js)

const image = document.getElementById("clickableImage");
const overlay = document.getElementById("tagOverlay");

// Sample tag data: you can load from server
const tags = [
  {
    username: "creator.design",
    xPercent: 60, // X = 60% from left
    yPercent: 40  // Y = 40% from top
  },
  {
    username: "utility.form",
    xPercent: 30,
    yPercent: 70
  }
];

// Function to inject tags into overlay
function renderTags(tagData) {
  overlay.innerHTML = ""; // clear previous

  tagData.forEach(tag => {
    const tagEl = document.createElement("div");
    tagEl.classList.add("tag");
    tagEl.textContent = `@${tag.username}`;
    tagEl.style.left = `${tag.xPercent}%`;
    tagEl.style.top = `${tag.yPercent}%`;
    tagEl.setAttribute("data-username", tag.username);

    tagEl.addEventListener("click", (e) => {
      e.stopPropagation(); // Prevent overlay toggle
      window.location.href = `/user/${tag.username}`;
    });

    overlay.appendChild(tagEl);
  });
}

// Toggle overlay on image click
let tagsVisible = false;

image.addEventListener("click", () => {
  tagsVisible = !tagsVisible;
  overlay.style.display = tagsVisible ? "block" : "none";
});

// Inject tags on load
renderTags(tags);

// Handle inline user tags in captions
document.querySelectorAll('.user-tag').forEach(tag => {
  tag.addEventListener('click', () => {
    const username = tag.textContent.replace('@', '');
    window.location.href = `/user/${username}`;
  });
});
